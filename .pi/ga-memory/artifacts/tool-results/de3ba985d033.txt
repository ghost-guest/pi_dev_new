  const input = usage?.input_tokens ?? 0;
  const output = usage?.output_tokens ?? 0;
  const cacheRead = usage?.cache_read_input_tokens ?? 0;
  const cacheWrite = usage?.cache_creation_input_tokens ?? 0;
  return { input, output, cacheRead, cacheWrite };
}

async function sleep(ms: number) {
  await new Promise((resolve) => setTimeout(resolve, ms));
}

function isTransientStatus(status: number) {
  return [502, 503, 504, 520, 522, 524].includes(status);
}

type AnyrouterStreamBlock = (ThinkingContent | TextContent | (ToolCall & { partialJson: string })) & { index: number };

function maybeNumber(value: unknown): number | undefined {
  return typeof value === "number" && Number.isFinite(value) ? value : undefined;
}

function applyUsage(model: Model<Api>, output: AssistantMessage, usage: unknown) {
  if (!usage || typeof usage !== "object") return;
  const source = usage as Record<string, unknown>;
  const input = maybeNumber(source.input_tokens);
  const outputTokens = maybeNumber(source.output_tokens);
  const cacheRead = maybeNumber(source.cache_read_input_tokens);
  const cacheWrite = maybeNumber(source.cache_creation_input_tokens);

  if (input !== undefined) output.usage.input = input;
  if (outputTokens !== undefined) output.usage.output = outputTokens;
  if (cacheRead !== undefined) output.usage.cacheRead = cacheRead;
  if (cacheWrite !== undefined) output.usage.cacheWrite = cacheWrite;
  output.usage.totalTokens = output.usage.input + output.usage.output + output.usage.cacheRead + output.usage.cacheWrite;
  calculateCost(model, output.usage);
}

function cleanupInternalBlockFields(output: AssistantMessage) {
  for (const block of output.content as AnyrouterStreamBlock[]) {
    delete (block as { index?: number }).index;
    delete (block as { partialJson?: string }).partialJson;
  }
}

function mapContentBlockIndex(output: AssistantMessage, index: number): number {
  return (output.content as AnyrouterStreamBlock[]).findIndex((block) => block.index === index);
}

function createIdleWatchdog(upstreamSignal?: AbortSignal): {
  signal: AbortSignal;
  reset: () => void;
  cleanup: () => void;
} {
  const configuredTimeoutMs = Number(process.env.ANYROUTER_IDLE_TIMEOUT_MS || 600000);
  const timeoutMs = Number.isFinite(configuredTimeoutMs) && configuredTimeoutMs > 0 ? configuredTimeoutMs : 600000;
  const controller = new AbortController();
  let timer: ReturnType<typeof setTimeout> | undefined;

  const abortFromUpstream = () => controller.abort(upstreamSignal?.reason || new Error("Request was aborted"));
  const reset = () => {
    if (!Number.isFinite(timeoutMs) || timeoutMs <= 0) return;
    if (timer) clearTimeout(timer);
    timer = setTimeout(() => {
      controller.abort(new Error(`Anyrouter stream idle timeout after ${timeoutMs}ms`));
    }, timeoutMs);
  };
  const cleanup = () => {
    if (timer) clearTimeout(timer);
    upstreamSignal?.removeEventListener("abort", abortFromUpstream);
  };

  if (upstreamSignal?.aborted) abortFromUpstream();
  else upstreamSignal?.addEventListener("abort", abortFromUpstream, { once: true });
  reset();

  return { signal: controller.signal, reset, cleanup };
}

function emitNonStreamingResponse(
  model: Model<Api>,
  stream: AssistantMessageEventStream,
  output: AssistantMessage,
  json: any,
  tools?: Tool[],
) {
  const blocks = Array.isArray(json?.content) ? json.content : [];

  for (const block of blocks) {
    if (block?.type === "text") {
      output.content.push({ type: "text", text: block.text || "" });
      const idx = output.content.length - 1;
      stream.push({ type: "text_start", contentIndex: idx, partial: output });
      if (block.text) stream.push({ type: "text_delta", contentIndex: idx, delta: block.text, partial: output });
      stream.push({ type: "text_end", contentIndex: idx, content: block.text || "", partial: output });
    } else if (block?.type === "thinking") {
      output.content.push({
        type: "thinking",
        thinking: block.thinking || "",
        thinkingSignature: block.signature,
      } as any);
      const idx = output.content.length - 1;
      stream.push({ type: "thinking_start", contentIndex: idx, partial: output });
      if (block.thinking) stream.push({ type: "thinking_delta", contentIndex: idx, delta: block.thinking, partial: output });
      stream.push({ type: "thinking_end", contentIndex: idx, content: block.thinking || "", partial: output });
    } else if (block?.type === "redacted_thinking") {
      output.content.push({
        type: "thinking",
        thinking: "[Reasoning redacted]",
        thinkingSignature: block.data,
        redacted: true,
      } as any);
      const idx = output.content.length - 1;
      stream.push({ type: "thinking_start", contentIndex: idx, partial: output });
      stream.push({ type: "thinking_end", contentIndex: idx, content: "[Reasoning redacted]", partial: output });
    } else if (block?.type === "tool_use") {
      const toolCall: ToolCall = {
        type: "toolCall",
        id: block.id,
        name: fromClaudeCodeToolName(block.name, tools),
        arguments: block.input || {},
      };
      output.content.push(toolCall as any);
      const idx = output.content.length - 1;
      stream.push({ type: "toolcall_start", contentIndex: idx, partial: output });
      stream.push({ type: "toolcall_delta", contentIndex: idx, delta: JSON.stringify(block.input || {}), partial: output });
      stream.push({ type: "toolcall_end", contentIndex: idx, toolCall, partial: output });
    }
  }

  applyUsage(model, output, json?.usage);
  output.stopReason = mapStopReason(json?.stop_reason);
}

function applyAnyrouterStreamEvent(
  model: Model<Api>,
  stream: AssistantMessageEventStream,
  output: AssistantMessage,
  event: any,
  tools?: Tool[],
) {
  if (!event || typeof event !== "object") return;

  if (event.type === "error") {
    throw new Error(event.error?.message || event.message || JSON.stringify(event));
  }

  if (event.type === "message_start") {
    if (typeof event.message?.id === "string") output.responseId = event.message.id;
    if (typeof event.message?.model === "string") output.responseModel = event.message.model;
    applyUsage(model, output, event.message?.usage);
    return;
  }

  if (event.type === "content_block_start") {
    if (event.content_block?.type === "text") {
      output.content.push({ type: "text", text: event.content_block.text || "", index: event.index } as any);
      const idx = output.content.length - 1;
      stream.push({ type: "text_start", contentIndex: idx, partial: output });
      if (event.content_block.text) {
        stream.push({ type: "text_delta", contentIndex: idx, delta: event.content_block.text, partial: output });
      }
    } else if (event.content_block?.type === "thinking") {
      output.content.push({
        type: "thinking",
        thinking: event.content_block.thinking || "",
        thinkingSignature: event.content_block.signature || "",
        index: event.index,
      } as any);
      const idx = output.content.length - 1;
      stream.push({ type: "thinking_start", contentIndex: idx, partial: output });
      if (event.content_block.thinking) {
        stream.push({ type: "thinking_delta", contentIndex: idx, delta: event.content_block.thinking, partial: output });
      }
    } else if (event.content_block?.type === "redacted_thinking") {
      output.content.push({
        type: "thinking",
        thinking: "",
        thinkingSignature: event.content_block.data || "",
        redacted: true,
        index: event.index,
      } as any);
      const idx = output.content.length - 1;
      stream.push({ type: "thinking_start", contentIndex: idx, partial: output });
    } else if (event.content_block?.type === "tool_use") {
      const input = event.content_block.input && typeof event.content_block.input === "object" ? event.content_block.input : {};
      output.content.push({
        type: "toolCall",
        id: event.content_block.id,
        name: fromClaudeCodeToolName(event.content_block.name, tools),
        arguments: input,
        partialJson: "",
        index: event.index,
      } as any);
      stream.push({ type: "toolcall_start", contentIndex: output.content.length - 1, partial: output });
    }
    return;
  }

  if (event.type === "content_block_delta") {
    const idx = mapContentBlockIndex(output, event.index);
    const block = (output.content as AnyrouterStreamBlock[])[idx];
    if (!block) return;

    if (event.delta?.type === "text_delta" && block.type === "text") {
      block.text += event.delta.text || "";
      if (event.delta.text) stream.push({ type: "text_delta", contentIndex: idx, delta: event.delta.text, partial: output });
    } else if (event.delta?.type === "thinking_delta" && block.type === "thinking") {
      block.thinking += event.delta.thinking || "";
      if (event.delta.thinking) {
        stream.push({ type: "thinking_delta", contentIndex: idx, delta: event.delta.thinking, partial: output });
      }
    } else if (event.delta?.type === "signature_delta" && block.type === "thinking") {
      block.thinkingSignature = (block.thinkingSignature || "") + (event.delta.signature || "");
    } else if (event.delta?.type === "input_json_delta" && block.type === "toolCall") {
      block.partialJson += event.delta.partial_json || "";
      try {
        block.arguments = block.partialJson ? JSON.parse(block.partialJson) : {};
      } catch {}
      if (event.delta.partial_json) {
        stream.push({ type: "toolcall_delta", contentIndex: idx, delta: event.delta.partial_json, partial: output });
      }
    }
    return;
  }

  if (event.type === "content_block_stop") {
    const idx = mapContentBlockIndex(output, event.index);
    const block = (output.content as AnyrouterStreamBlock[])[idx];
    if (!block) return;

    delete (block as { index?: number }).index;
    if (block.type === "text") {
      stream.push({ type: "text_end", contentIndex: idx, content: block.text, partial: output });
    } else if (block.type === "thinking") {
      stream.push({ type: "thinking_end", contentIndex: idx, content: block.thinking, partial: output });
    } else if (block.type === "toolCall") {
      try {
        block.arguments = block.partialJson ? JSON.parse(block.partialJson) : block.arguments;
      } catch {}
      delete (block as { partialJson?: string }).partialJson;
      stream.push({ type: "toolcall_end", contentIndex: idx, toolCall: block, partial: output });
    }
    return;
  }

  if (event.type === "message_delta") {
    if (event.delta?.stop_reason) output.stopReason = mapStopReason(event.delta.stop_reason);
    applyUsage(model, output, event.usage);
  }
}

async function consumeAnyrouterBody(
  model: Model<Api>,
  response: Response,
  stream: AssistantMessageEventStream,
  output: AssistantMessage,
  resetIdleTimer: () => void,
  tools?: Tool[],
): Promise<boolean> {
  if (!response.body) throw new Error("Anyrouter response has no body");

  const decoder = new TextDecoder();
  let buffer = "";
  let rawBody = "";
  let sawSseEvent = false;

  const handleFrame = (frame: string) => {
    const data = frame
      .split(/\r?\n/)
      .filter((line) => line.startsWith("data:"))
      .map((line) => line.slice(5).trimStart())
      .join("\n");
    if (!data || data === "[DONE]") return;
    sawSseEvent = true;
    const event = JSON.parse(data);
    logAnyrouter("stream_event", { type: event.type });
    applyAnyrouterStreamEvent(model, stream, output, event, tools);
  };

  for await (const chunk of response.body as any) {
    resetIdleTimer();
    const text = decoder.decode(chunk, { stream: true });
    rawBody += text;
    buffer += text;

    const frames = buffer.split(/\r?\n\r?\n/);
    buffer = frames.pop() || "";
    for (const frame of frames) handleFrame(frame);
  }

  const tail = decoder.decode();
  if (tail) {
    rawBody += tail;
    buffer += tail;
  }
  if (buffer.trim()) handleFrame(buffer);

  if (!sawSseEvent) {
    logAnyrouter("non_sse_response", { body: rawBody.slice(0, 4000) });
    emitNonStreamingResponse(model, stream, output, JSON.parse(rawBody), tools);
  }

  return sawSseEvent;
}

async function requestAnyrouterStream(
  model: Model<Api>,
  context: Context,
  options: SimpleStreamOptions | undefined,
  stream: AssistantMessageEventStream,
  output: AssistantMessage,
) {
  const apiKey = options?.apiKey || "";
  const url = `${String(model.baseUrl || DEFAULT_BASE_URL).replace(/\/$/, "")}/v1/messages?beta=true`;
  const payload = buildPayload(model, context, options);
  const headers = {
    ...buildHeaders(apiKey),
    ...(options?.headers || {}),
  };

  const configuredMaxAttempts = Number(process.env.ANYROUTER_MAX_ATTEMPTS || 1);
  const maxAttempts = Number.isFinite(configuredMaxAttempts) && configuredMaxAttempts > 0 ? Math.trunc(configuredMaxAttempts) : 1;
  let lastError: Error | undefined;
  let sawSseEvent = false;

  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    const watchdog = createIdleWatchdog(options?.signal);
    logAnyrouter("request", {
      attempt,
      url,
      model: payload.model,
      stream: payload.stream,
      thinking: payload.thinking,
      max_tokens: payload.max_tokens,
      headerKeys: Object.keys(headers),
    });

    try {
      const response = await fetch(url, {
        method: "POST",
        headers,
        body: JSON.stringify(payload),
        signal: watchdog.signal,
      });

      logAnyrouter("response_headers", {
        attempt,
        status: response.status,
        ok: response.ok,
        contentType: response.headers.get("content-type"),
      });

      if (!response.ok) {
        const text = await response.text();
        logAnyrouter("response_error", { attempt, status: response.status, body: text.slice(0, 4000) });
        if (isTransientStatus(response.status) && attempt < maxAttempts) {
          await sleep(1000 * attempt);
          continue;
        }
        throw new Error(`HTTP ${response.status}: ${text}`);
      }

      sawSseEvent = await consumeAnyrouterBody(model, response, stream, output, watchdog.reset, context.tools);
      return;
    } catch (error) {
      lastError = error instanceof Error ? error : new Error(String(error));
      logAnyrouter("error", { attempt, message: lastError.message, sawSseEvent });
      if (options?.signal?.aborted || watchdog.signal.aborted || sawSseEvent) throw lastError;
      if (attempt < maxAttempts) {
        await sleep(1000 * attempt);
        continue;
      }
    } finally {
      watchdog.cleanup();
    }
  }

  throw lastError || new Error("Anyrouter request failed");
}

function streamAnyrouterClaude(
  model: Model<Api>,
  context: Context,
  options?: SimpleStreamOptions,
): AssistantMessageEventStream {
  const stream = createAssistantMessageEventStream();

  (async () => {
    const output: AssistantMessage = {
      role: "assistant",
      content: [],
      api: model.api,
      provider: model.provider,
      model: model.id,
      usage: {
        input: 0,
        output: 0,
        cacheRead: 0,
        cacheWrite: 0,
        totalTokens: 0,
        cost: { input: 0, output: 0, cacheRead: 0, cacheWrite: 0, total: 0 },
      },
      stopReason: "stop",
      timestamp: Date.now(),
    };

    try {
      stream.push({ type: "start", partial: output });
      await requestAnyrouterStream(model, context, options, stream, output);
      cleanupInternalBlockFields(output);
      if (output.stopReason === "error") throw new Error("Anyrouter returned an error stop reason");
      stream.push({ type: "done", reason: output.stopReason as "stop" | "length" | "toolUse", message: output });
      stream.end();
    } catch (error) {
      cleanupInternalBlockFields(output);
      const message = error instanceof Error ? error.message : String(error);
      output.stopReason = options?.signal?.aborted || message.includes("Anyrouter stream idle timeout") ? "aborted" : "error";
      output.errorMessage = message;
      stream.push({ type: "error", reason: output.stopReason, error: output });
      stream.end();
    }
  })();

  return stream;
}
