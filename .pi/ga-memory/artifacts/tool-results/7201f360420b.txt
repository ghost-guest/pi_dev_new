175: def expand_file_refs(text, base_dir=None):
176:     """展开文本中的 {{file:路径:起始行:结束行}} 引用为实际文件内容。

179:     pattern = r'\{\{file:(.+?):(\d+):(\d+)\}\}'
180:     def replacer(match):
181:         path, start, end = match.group(1), int(match.group(2)), int(match.group(3))
182:         path = os.path.abspath(os.path.join(base_dir or '.', path))
183:         if not os.path.isfile(path): raise ValueError(f"引用文件不存在: {path}")
184:         with open(path, 'r', encoding='utf-8') as f: lines = f.readlines()
185:         if start < 1 or end > len(lines) or start > end: raise ValueError(f"行号越界: {path} 共{len(lines)}行, 请求{start}-{end}")
186:         return ''.join(lines[start-1:end])
187:     return re.sub(pattern, replacer, text)
